How Repos Adopt Shared Workflows

Consumer adds ~15 lines

# .github/workflows/ci.yml
name: CI
on: [push, pull_request]

jobs:
  ci:
    uses: mruthyunjaya-lakkappanavar/
      github-shared-workflows/
      .github/workflows/reusable-ci.yml@main
    with:
      language: python
      language_version: "3.12"
      enable_lint: true
      enable_security_scan: true
    secrets:
      SLACK_WEBHOOK_URL: ${{ secrets... }}
